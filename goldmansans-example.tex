\documentclass{article}
\usepackage[]{goldmansans}
\usepackage{blindtext}
\begin{document}
   \blinddocument
   
   \textsl{Test} vs. \textit{Test}
   
   {\goldmansansCd Test}

   {\goldmansansCd\bfseries Test}

   {\goldmansansBlk Test}   
   
   {\goldmansansLt Test}   
   
   {\goldmansansTh Test}   

   {\goldmansansSb Test}   
   
   {Test \strong{Test \strong{Test}}}
   
   ff fi fj ffi ffj ij vs. {\addfontfeatures{Ligatures=NoCommon}ff fi fj ffi ffj ij}
   
   {\goldmansansfrac These are fractions: 1/2 and 3/6} vs. These are not fractions: 1/2 and 3/6
   
   Test\textsuperscript{2} vs. Test\textsuperscript*{2}
   
   1 2 3 4 5 6 7 8 9 0 vs. {\goldmansansTf 1 2 3 4 5 6 7 8 9 0}
\end{document}
